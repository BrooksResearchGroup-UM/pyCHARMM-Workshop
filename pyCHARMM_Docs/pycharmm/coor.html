129def copy_forces(mass=False, selection=None):  
130    """Copy the forces to the comparison set  
131  
132    Parameters
133    ----------
134    mass : bool, default = False    
135        Whether to weight the forces by mass  
136    selection : pycharmm.SelectAtoms    
137        only copy forces of selected atoms

142        true if successful  
143    """

149

415def orient(**kwargs):  
416    """Modifies coordinates of all atoms according to the passed flags  
417  
418    The select set of atoms is first centered about the origin,  
419    and then rotated to either align with the axis,  
420    or the other coordinate set.  
421  
422    The *RMS* keyword will use the other coordinate set as a rotation reference.  
423  
424    The *MASS* keyword cause a mass weighting to be done. This will  
425    align the specified atoms along their moments of inertia. When the RMS  
426    keyword is not used, then the structure is rotated so that its principle  
427    geometric axis coincides with the X-axis and the next largest coincides  
428    with the Y-axis. This command is primarily used for preparing a  
429    structure for graphics and viewing. It can also be used for finding  
430    RMS differences, and in conjunction with the vibrational analysis.  
431  
432    The *NORO*tation keyword will suppress rotations. In this case,  
433    only one coordinate set will be modified.  
434  
435    Parameters
436    ----------
437    **kwargs : dictionary  
438        key words for the coor orient CHARMM command  
439    """  
440    oscript = pycharmm.script.CommandScript('coor', orient=True, **kwargs)  
441    oscript.run()

471def stat(selection=None, comp=False, mass=False):  
472    """Computes *max*, *min*, *ave* for `x`, `y`, `z`, `w` over selection of main or comp sets  
473  
474    Parameters
475    ----------
476    selection : pycharmm.SelectAtoms    
477        a selection of atom indexes to use for stats  
478    comp : bool    
479        if true, stats computed for comparison set selection  
480    mass : bool    
481        if true, will place the average values at the center of mass  
482
483    Returns
484    -------
485    dict
486        A python dictionary with keys   
487        xmin, xmax, xave,   
488        ymin, ymax, yave,   
489        zmin, zmax, zave,   
490        wmin, wmax, wave,   
491        n_selected, n_misses  
492    """  
493    c_comp = ctypes.c_int(comp)  
494    c_mass = ctypes.c_int(mass)  
495  
496    if selection is None:  
497        selection = pycharmm.SelectAtoms().all_atoms()  
498  
499    c_sel = selection.as_ctypes()  
500  
501    n_stats = 13  
502    max_label = 5  
503  
504    vals = (ctypes.c_double * n_stats)()  
505  
506    labels_bufs = [ctypes.create_string_buffer(max_label)  
507                   for _ in range(n_stats)]  
508    labels_ptrs = (ctypes.c_char_p * n_stats)(*map(ctypes.addressof,  
509                                                   labels_bufs))  
510  
511    n_sel = ctypes.c_int(0)  
512    n_misses = ctypes.c_int(0)  
513  
514    lib.charmm.coor_stat(c_sel,  
515                         ctypes.byref(c_comp), ctypes.byref(c_mass),  
516                         labels_ptrs, vals,  
517                         ctypes.byref(n_sel), ctypes.byref(n_misses))  
518  
519    labels_str = [label.value.decode(errors='ignore')  
520                  for label in labels_bufs[0:n_stats]]  
521    labels = [label.strip().lower() for label in labels_str]  
522    labels = [label for label in labels if label]  
523  
524    stats = dict(zip(labels, vals))  
525    stats['n_selected'] = n_sel.value  
526    stats['n_misses'] = n_misses.value  
527    return stats

533class Coordinates:  
534    __slots__ = 'which_set', 'coords', 'are_dirty'  
535  
536    def __init__(self, coord_set):  
537        self.which_set = coord_set  
538        self.coords = pandas.DataFrame(columns=['x', 'y', 'z', 'w'])  
539        self.are_dirty = True  
540        self.pull()  
541  
542    def pull(self):  
543        if self.which_set is 'main':  
544            self.coords = get_main()  
545        elif self.which_set is 'comp':  
546            self.coords = get_comparison()  
547        elif self.which_set is 'comp2':  
548            self.coords = get_comp2()  
549        else:  
550            msg = '{} is not a valid coordinate set'.format(self.which_set)  
551            raise ValueError(msg)  
552  
553        self.are_dirty = False  
554        return self  
555  
556    def push(self):  
557        if self.which_set is 'main':  
558            set_main(self.coords)  
559        elif self.which_set is 'comp':  
560            set_comparison(self.coords)  
561        elif self.which_set is 'comp2':  
562            set_comp2(self.coords)  
563        else:  
564            msg = '{} is not a valid coordinate set'.format(self.which_set)  
565            raise ValueError(msg)  
566  
567        self.are_dirty = False  
568        return self  
569  
570    def __len__(self):  
571        return len(self.coords)

472def stat(selection=None, comp=False, mass=False):

474
