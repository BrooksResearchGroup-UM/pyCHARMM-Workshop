  83def get_charges():
  84    """Get charges for the atoms
  85
  86    Returns
  87    -------
  88    charges : list[float]
  89        charges for the atoms
  90    """
  91    n = psf.get_natom()
  92    charges = (ctypes.c_double * n)(0)
  93    status = lib.charmm.scalar_get_charges(charges)
  94    qstatus = bool(status)
  95    if not qstatus:
  96        raise RuntimeError('There was a problem getting charges.')
  97
  98    return list(charges)

 101def set_charges(charges, selection=None):
 102    """Set charges for the selected atoms
 103
 104    Parameters
 105    ----------
 106    charges : list[float]
 107        charges for each atom
 108    selection: pycharmm.SelectAtoms
 109        set charges for only these selected atoms
 110
 111        default None results in all atoms selected
 112
 113    Returns
 114    -------
 115    int
 116        status code returned by lib.charmm.scalar_set_charges routine
 117    """
 118    n = psf.get_natom()
 119    charges = (ctypes.c_double * n)(*charges)
 120
 121    if selection is None:
 122        selection = pycharmm.SelectAtoms().all_atoms()
 123
 124    c_sel = selection.as_ctypes()
 125
 126    status = lib.charmm.scalar_set_charges(charges, c_sel)
 127    qstatus = bool(status)
 128    if not qstatus:
 129        raise RuntimeError('There was a problem setting charges.')
 130
 131    return int(status)
