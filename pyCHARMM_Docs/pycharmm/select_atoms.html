pycharmm.select_atoms

Functions for selecting atoms in a structure

Corresponds to CHARMM command SELEction

See CHARMM documentation select for more information

Examples

>>> import pycharmm

Select all atoms in a protein, whose segment ID is PROT

>>> sele_prot = pycharmm.SelectAtoms(seg_id='PROT')

Select all CA atoms

>>> sele_ca = pycharmm.SelectAtoms(atom_type='CA')

Select all CA atoms of the protein

>>> sele_prot_ca = sele_prot & sele_ca

Save the above selection in CHARMM with name prot_ca. This is similar to CHARMM command:

define prot_ca sele segid PROT .and. type CA end

>>> sele_prot_ca.store('prot_ca')

For segments ALAD and GLAD, select all atoms whose residue IDs are either 1 or 2, and whose atom names are either N or CA

>>> atoms = pycharmm.SelectAtoms().by_res_and_type('ALAD GLAD','1 2','N CA')

 45def _get_random_string(str_size):
 46    return ''.join(random.choice(string.ascii_letters)
 47                   for _ in range(str_size)).upper()

 50def _generate_script_name():
 51    max_name = select.get_max_name()
 52    rand_name = _get_random_string(max_name)
 53    found = select.find(rand_name)
 54    iter_limit = 100
 55    while found > 0 and iter_limit > 0:
 56        rand_name = _get_random_string(max_name)
 57        found = select.find(rand_name)
 58        iter_limit -= 1
 59
 60    if iter_limit <= 0 < found:
 61        raise RuntimeError('For this selection, ' +
 62                           'tried 100 random names, and ' +
 63                           'could not find an unused random name.')
 64
 65    return rand_name

 69    def __init__(self, selection=None,
 70                 select_all=False,
 71                 seg_id='',
 72                 res_id='', res_name='',
 73                 atom_nums=None,
 74                 atom_type='', chem_type='',
 75                 initials=False,
 76                 lonepairs=False,
 77                 hydrogens=False,
 78                 update=True):
 79        """
 80        Parameters
 81        ----------
 82        selection : bool tuple
 83            A boolean tuple whose length is the number of atoms
 84            in the system. True if the corresponding atom is selected.
 85        select_all : bool
 86            True for selecting all atoms
 87        seg_id : str
 88            segment ID of the selection
 89        res_id : str
 90            residue ID of the selection
 91        res_name : str
 92            residue name of the selection
 93        atom_nums : int or list of int
 94            atom index, i.e., atom number 
 95        atom_type : str
 96            atom names of the selection
 97        chem_type : str
 98            chemical type of the selection
 99        initials : bool
100            True for selecting all atoms with known coordinates
101        lonepairs : bool
102            True for selecting all lone pairs
103        hydrogens : bool
104            True for selecting all hydrogen atoms
105        update : bool
106            True for updating the selection
107            
108        """
109        self._name = ''
110        self._stored = False
111        self._do_update = update
112        self._selection = None
113        n_atoms = psf.get_natom()
114        if selection:
115            self.set_selection(selection)
116        else:
117            self.set_selection(select.none_selection(n_atoms))
118
119        if select_all:
120            self.set_selection(select.all_selection(n_atoms))
121
122        if seg_id:
123            self.by_seg_id(seg_id)
124
125        if res_id:
126            self.by_res_id(res_id)
127
128        if res_name:
129            self.by_res_name(res_name)
130        
131        if not atom_nums is None:
132            if isinstance(atom_nums, int):
133                self.by_atom_nums([atom_nums])
134            else:
135                self.by_atom_nums(atom_nums)
136        
137        if atom_type:
138            self.by_atom_type(atom_type)
139
140        if chem_type:
141            self.by_chem_type(chem_type)
142
143        if initials:
144            self.all_initial_atoms()
145
146        if lonepairs:
147            self.all_lonepair_atoms()
148
149        if hydrogens:
150            self.all_hydrogen_atoms()

153        old_val = self[atom_i]

158        old_val = self[atom_i]
159        new_sel = tuple(sel if not ind == atom_i else False

275    def in_sphere(self, x, y, z, radius=0.8, is_periodic=False):
276        """ Select all atoms within a sphere around point (x,y,z) with a certain radius.
277
278        Corresponds to CHARMM command `sele point`
279
280        Parameters
281        ----------
282        x : float
283              x coordinate of the reference point
284        y : float
285              y coordinate of the reference point
286        z : float
287              z coordinate of the reference point
288        radius : float 
289              radius of the shpere around a given point
290        is_periodic : bool
291              If True AND simple periodic boundary conditions are in effect
292              through the use of the MIPB command, the selection reflects
293              the appropriate periodic boundaries.
294              see [images](<https://academiccharmm.org/documentation/version/c47b1/images/>)
295        """
296        new_sel = select.or_selection(self.get_selection(),
297                                      select.by_point(x, y, z,
298                                                      radius,
299                                                      is_periodic))
300        self.set_selection(new_sel)
301        return self

303    def by_res_and_type(self, seg_id, res_id, atom_type):
304        """Select multiple segments, resids and atom types. Specifically,
305        for all segments whose names are in `seg_id`, select all atoms whose
306        residue IDs are in `res_id` and whose atom names are in `atom_type`.
307
308        Parameters
309        ----------
310        seg_id : string
311                 string of segment identifiers ('A1 MAIN SEG1 ... SEGn')
312        res_id : string
313                 string of residue identifiers ('1 3 5 6 ...n')
314        atom_type : string
315                    string of an IUPAC names ('C CA CB N S')
316
317        Returns
318        -------
319        flags : boolean list
320                atom i selected <==> flags[i] == True
321        """
322        seg = SelectAtoms()
323        for segid in seg_id.strip().split():
324            seg.by_seg_id(segid)
325        res = SelectAtoms()
326        for resid in res_id.strip().split():
327            res.by_res_id(res_id=resid)
328        atoms = SelectAtoms()
329        for atomname in atom_type.strip().split():
330            atoms.by_atom_type(atom_type=atomname)
331        my_atom = seg & res & atoms
332        new_sel = select.or_selection(self.get_selection(),
333                                      my_atom.get_selection())
334        self.set_selection(new_sel)
335        return self

407    def as_ctypes(self):
408        """
409        For a pycharmm selection, convert it to a selection 
410        for lib.charmm to use
411        """
412        atoms = [int(atom) for atom in list(self)]
413        natoms = len(atoms)
414        c_selection = (ctypes.c_int * natoms)(*atoms)

488    def store(self, name=''):
489        """
490        Save the pycharmm selection to CHARMM 
491
492        Parameters
493        ----------
494        name : str
495            name of pycharmm selection in CHARMM
496        """
497        max_name = select.get_max_name()
498        if name:
499            self._name = name[:max_name]
500
501        if not self._name:
502            self._name = _generate_script_name()
503
504        if len(self._name) > max_name:
505            self._name = self._name[:max_name]
506
507        select.store_selection(self._name, self.get_selection())
508        self._stored = True
509        return self._name
510
511    def unstore(self):
512        """
513        Unstore/remove the named pycharmm selection in CHARMM
514
515        Returns
516        -------
517        was_stored : old store status
518        """
519        was_stored = self.is_stored()
520        if was_stored:
521            select.delete_stored_selection(self._name.upper())
522            self._stored = False
523
524        return was_stored

533    def _update(self):
534        sel = self.get_selection()
535        inds = [i for i, x in enumerate(sel) if x]
536        self._atom_indexes = inds
537        self._n_selected = len(inds)
538        self._chem_types = atom_info.get_chem_types(inds)
539        self._res_indexes = atom_info.get_res_indexes(inds)
540        self._res_names = atom_info.get_res_names(inds)
541        self._res_ids = atom_info.get_res_ids(inds)
542        self._seg_indexes = atom_info.get_seg_indexes(inds)
543        self._seg_ids = atom_info.get_seg_ids(inds)
544        self._atom_types = atom_info.get_atom_types(inds)
545        if self.is_stored():
546            self.store()
547
548        return self

604    def __init__(self, select_atoms):
605        self._select_atoms = select_atoms
606        self._index = 0

608    def __next__(self):
609        if self._index < len(self._select_atoms):
610            is_sel_i = self._select_atoms.is_selected(self._index)
611            self._index += 1
612            return is_sel_i

614        raise StopIteration
