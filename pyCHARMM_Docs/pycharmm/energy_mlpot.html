 28class MLpot():
 29    """
 30    Custom Machine Learning potential
 31
 32    """
 33    
 34    def __init__(
 35        self,
 36        # ML atom selection
 37        ml_selection,
 38        # Fluctuating ML atom charges for ML/MM electrostatic interaction
 39        ml_fq=True,
 40        # Additional model keyword arguments
 41        **kwargs):
 42        """    
 43        Parameters
 44        ----------
 45        ml_selection : pycharmm selection
 46             ML atom selection
 47
 48        ml_fq : bool
 49             Fluctuating ML atom charges for ML/MM electrostatic interaction
 50
 51        **kwargs : dict
 52             Additional model keyword arguments
 53
 54        """    
 55        # ML&MM - atom number
 56        self.Natoms = pycharmm.coor.get_natom()
 57        
 58        # ML atoms - atom indices
 59        ml_indices = ml_selection.get_atom_indexes()
 60        ml_argsort = np.argsort(np.array(ml_indices, dtype=int))
 61        self.ml_indices = np.array(ml_indices, dtype=int)[ml_argsort]
 62        
 63        # ML atoms - atom number
 64        self.ml_Natoms = len(ml_indices)
 65        
 66        # ML - Delete bonds, angles, dihedrals, improper - if active
 67        pycharmm.psf.delete_bonds(ml_selection, ml_selection)
 68        pycharmm.psf.delete_angles(ml_selection, ml_selection)
 69        pycharmm.psf.delete_dihedrals(ml_selection, ml_selection)
 70        pycharmm.psf.delete_impropers(ml_selection, ml_selection)
 71        
 72        # ML&MM - psf charges - set ML charges zero when charges and ML-MM 
 73        # interaction are handled by the ML potential
 74        self.ml_fq = ml_fq
 75        self.mlmm_charges = np.array(pycharmm.param.get_charge())
 76        if ml_fq:
 77            self.mlmm_charges[self.ml_indices] = 0.0
 78            _ = pycharmm.psf.set_charge(self.mlmm_charges)
 79        
 80        # ML - set non-bond exclusion list for ML atom pairs
 81        self.ml_iblo = np.zeros(self.Natoms, dtype=int)
 82        self.ml_inb = []
 83        for ii, idx in enumerate(ml_indices):
 84            self.ml_iblo[idx:] += self.ml_Natoms - ii - 1
 85            for jdx in self.ml_indices[(ii + 1):]:
 86                self.ml_inb.append(jdx + 1) # + 1 as CHARMM start at index 1
 87        self.ml_nnb = len(self.ml_inb)
 88        
 89        pycharmm.psf.set_iblo_inb(self.ml_iblo, self.ml_inb)
 90        pycharmm.nbonds.update_bnbnd() # Already executed in set_iblo_inb()
 91        pycharmm.image.update_bimag()
 92        
 93        ###################################################
 94        # START - Potential model dependent part
 95        #||||||||||||||||||||||||||||||||||||||||||||||||||
 96        #VVVVVVVVVVVVVVVVVVVVVVVVVVVVVVVVVVVVVVVVVVVVVVVVVV
 97        
 98        # Import modules just at initialization as tensorflow would be 
 99        # initialized all the time
100        from PhysNet_CHARMM import PhysNet_Calc
101        
102        # ML atoms - atomic numbers
103        if "Z" in kwargs.keys():
104            ml_Z = np.array(kwargs["Z"], dtype=int)[ml_argsort]
105        else:
106            raise IOError("ML atom number 'Z' for the selected atoms is mandatory!")
107        # Checkpoint file to restore the model 
108        if "checkpoint" in kwargs.keys():
109            checkpoint = kwargs["checkpoint"]
110        else:
111            raise IOError("Tensorflow checkpoint path 'checkpoint' is mandatory!")
112        # Respective config file for PhysNet architecture
113        if "config" in kwargs.keys():
114            config = kwargs["config"]
115        else:
116            raise IOError("PhysNet config file 'config' is mandatory!")
117        # System charge
118        if "charge" in kwargs.keys():
119            ml_charge = kwargs["charge"]
120        else:
121            ml_charge = 0
122        # ML/MM electrostatic interaction max cutoff
123        if "ctofnb" in kwargs.keys():
124            ctofnb = kwargs["ctofnb"]
125        else:
126            ctofnb = None
127        # ML/MM electrostatic interaction min cutoff
128        if "ctonnb" in kwargs.keys():
129            ctonnb = kwargs["ctonnb"]
130        else:
131            ctonnb = None
132        # Energy conversion factor
133        if "econv" in kwargs.keys():
134            econv = kwargs["econv"]
135        else:
136            econv = 1.0
137        # Force conversion factor
138        if "fconv" in kwargs.keys():
139            fconv = kwargs["fconv"]
140        else:
141            fconv = 1.0
142        # Checkpoint file fron PhysNet v1?
143        if "v1" in kwargs.keys():
144            v1 = kwargs["v1"]
145        else:
146            v1 = False
147        
148        # PhysNet default parameter
149        self.params = {
150            'num_features': 128,
151            'num_basis': 64,
152            'num_blocks': 5,
153            'num_residual_atomic': 2,
154            'num_residual_interaction': 3,
155            'num_residual_output': 1,
156            'cutoff': 10.0,
157            'lr_cutoff': None,
158            'use_electrostatic': 1,
159            'use_dispersion': 1,
160            'grimme_s6': None,
161            'grimme_s8': None,
162            'grimme_a1': None,
163            'grimme_a2': None,
164            }
165        
166        # Read config file or dictionary
167        if isinstance(config, str):
168            
169            with open(config, 'r') as f:
170                config_lines = f.readlines()
171            
172            # Get PhysNet parameter
173            custom_params = {}
174            for cline in config_lines:
175                
176                # Read line
177                cpar = cline.split("=")
178                if len(cpar)!=2:
179                    print("Incorrect config line '{:s}' - skipped!". format(cline))
180                    print("Expect '--key item' format.")
181                    continue
182                else:
183                    (key, item) = cpar
184                
185                # Check key format
186                if "--" == key[:2]:
187                    key = key[2:]
188                else:
189                    print("Incorrect key format '{:s}' - skipped!". format(key))
190                    print("Expect '--key item' format.")
191                    continue
192                
193                # Check key duplicity:
194                if key in custom_params.keys():
195                    print("Multiple definitions of '{:s}' - skipped!". format(key))
196                    continue
197                
198                # Append custom parameter
199                custom_params[key] = item
200                
201        else:
202            
203            custom_params = config
204            
205        # Update PhysNet parameters
206        self.params.update(custom_params)
207        
208        # ML Interaction range
209        # sr_cutoff: short range cutoff (NN + elec + disp)
210        # lr_cutoff: long range cutoff (elec + disp)
211        if self.params['lr_cutoff'] is None:
212            sr_cutoff = float(self.params['cutoff'])
213            lr_cutoff = None
214        else:
215            sr_cutoff = float(self.params['cutoff'])
216            lr_cutoff = float(self.params['lr_cutoff'])
217            
218        # ML/MM interaction range
219        if ctofnb is None:
220            ctofnb = pycharmm.nbonds.get_ctofnb()
221        else:
222            ctofnb = float(ctofnb)
223        if ctonnb is None:
224            ctonnb = pycharmm.nbonds.get_ctonnb()
225        else:
226            ctonnb = float(ctonnb)
227        
228        if self.params['grimme_s6'] is None:
229            s6 = None
230        else:
231            s6 = float(self.params['grimme_s6'])
232        if self.params['grimme_s8'] is None:
233            s8 = None
234        else:
235            s8 = float(self.params['grimme_s8'])
236        if self.params['grimme_a1'] is None:
237            a1 = None
238        else:
239            a1 = float(self.params['grimme_a1'])
240        if self.params['grimme_a2'] is None:
241            a2 = None
242        else:
243            a2 = float(self.params['grimme_a2'])
244        
245        # Initialize PhysNet model
246        self.model = PhysNet_Calc(
247            self.Natoms,
248            self.ml_indices,
249            ml_Z,
250            self.ml_fq,
251            self.mlmm_charges,
252            ml_charge,
253            sr_cutoff,
254            lr_cutoff,
255            ctofnb,
256            ctofnb - ctonnb,
257            econv,
258            fconv,
259            int(self.params['num_features']),
260            int(self.params['num_basis']),
261            num_blocks=int(self.params['num_blocks']),
262            num_residual_atomic=int(self.params['num_residual_atomic']),
263            num_residual_interaction=int(self.params['num_residual_interaction']),
264            num_residual_output=int(self.params['num_residual_output']),
265            use_electrostatic=bool(self.params['use_electrostatic']),
266            use_dispersion=bool(self.params['use_dispersion']),
267            s6=s6,
268            s8=s8,
269            a1=a1,
270            a2=a2,
271            name="PhysNet"
272            )
273            
274        # Load PhysNet checkpoint file
275        self.checkpoint = checkpoint
276        if v1:
277            self.model.restore_v1(self.checkpoint)
278        else:
279            self.model.restore(self.checkpoint)
280        
281        # Initialize custom energy function
282        self.func_type = ctypes.CFUNCTYPE(
283            ctypes.c_double,                    # user energy - E(user)
284            ctypes.c_int,                       # Atom number central cell - Natom
285            ctypes.c_int,                       # Number of central and image cells - Ntrans
286            ctypes.c_int,                       # Atom number central + image cells - Natim
287            ctypes.POINTER(ctypes.c_double),    # x: Atom position x
288            ctypes.POINTER(ctypes.c_double),    # y: Atom position y
289            ctypes.POINTER(ctypes.c_double),    # z: Atom position y
290            ctypes.POINTER(ctypes.c_double),    # dx: Atom potential derivative dE/dx
291            ctypes.POINTER(ctypes.c_double),    # dy: Atom potential derivative dE/dy
292            ctypes.POINTER(ctypes.c_double),    # dz: Atom potential derivative dE/dz
293            ctypes.POINTER(ctypes.c_int),       # IMATTR: Corresponding central atom index of image atom
294            ctypes.c_int,                       # Nmlp: Number of ML-ML atom pairs
295            ctypes.c_int,                       # Nmlmmp: Number of ML-MM atom pairs
296            ctypes.POINTER(ctypes.c_int),       # idxi: ML-ML atom pair (atom i)
297            ctypes.POINTER(ctypes.c_int),       # idxj: ML-ML atom pair (atom j)
298            ctypes.POINTER(ctypes.c_int),       # idxu: ML-MM atom pair (ML atom u)
299            ctypes.POINTER(ctypes.c_int),       # idxv: ML-MM atom pair (MM atom v)
300            ctypes.POINTER(ctypes.c_int)        # idxp: Image to central MM atom pointer)
301            )
302        
303        self.energy_func = self.func_type(self.model.charmm_energy_and_forces)
304        
305        #AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA
306        #||||||||||||||||||||||||||||||||||||||||||||||||||
307        # END - Potential model dependent part
308        ###################################################
309        
310        pycharmm.lib.charmm.mlpot_set_func(self.energy_func)
311        
312        Nml = self.ml_Natoms
313        mlidx = (ctypes.c_int * Nml)(*(self.ml_indices + 1))
314        mlidz = (ctypes.c_int * Nml)(*ml_Z)
315        Nml = (ctypes.c_int * 1)(Nml)
316        pycharmm.lib.charmm.mlpot_set_properties(
317            Nml, mlidx, mlidz)
318        
319        self.is_set = True
320        
321    def __del__(self):
322        """class destructor
323        """
324        self.unset_mlpot()
325
326    def unset_mlpot(self):
327        """Just store the function and do not run it during energy calcs
328        """
329        pycharmm.lib.charmm.mlpot_unset()
330        self.is_set = False
