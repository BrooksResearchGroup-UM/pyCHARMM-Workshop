 10class CommandScript:
 11    """ A class used to construct a command in CHARMM scripting language and then execute it.
 12
 13    """
 14    def __init__(self, command, selection=None, **kwargs):
 15        """
 16        Parameters
 17        ----------
 18        command : str
 19            CHARMM scripting language command that will be run using pyCHARMM.
 20        selection : pycharmm.selectAtoms
 21        **kwargs : dict, optional
 22            Additional options relevant to the `command` being run.
 23        """
 24        self.command = str(command)
 25        self.selection = selection
 26        self.opts = dict()
 27        self.add_options(**kwargs)
 28
 29    def add_options(self, **kwargs):
 30        """Add additional options relevant to the command being run.
 31
 32        If value is of type `float`, `int` or `str` a "key value" 
 33        line is added to the command's script. 
 34        If it is of type `bool` and is True
 35        then a "key" line is added.
 36        """ 
 37        for k, v in kwargs.items():
 38            if type(v) in [float, int, str]:   # add continuation
 39                self.opts[k] = str(k) + ' ' + str(v) + ' -\n'
 40            elif (type(v) == bool):  # add continuation
 41                if v:
 42                    self.opts[k] = str(k) + ' -\n'
 43            else:
 44                message = 'invalid option {} = {}'
 45                raise ValueError(message.format(k, v))
 46
 47    def _add_selection(self, selection):
 48        if not selection.is_stored():
 49            raise ValueError('store selection before adding to script')
 50
 51        self.add_options(sele=selection.get_stored_name() + ' end')
 52        return
 53
 54    def _remove_selection(self, selection):
 55        self.opts.pop('sel', None)
 56        selection.unstore()
 57
 58    def create_script_string(self):
 59        script = self.command
 60        if self.opts.values():
 61            script += ' ' + ' '.join(self.opts.values())
 62            script = script.strip()
 63            if script.endswith('-'):
 64                script = script[:-1].strip()
 65            if not script.endswith('\n'):
 66                script += '\n'
 67
 68        return script
 69
 70    def run(self, append=''):
 71        """Execute the command.
 72        """
 73        if self.selection:
 74            selection = self.selection.get_selection()
 75            with atoms.SelectAtoms(selection=selection) as sel:
 76                self._add_selection(sel)
 77                script = self.create_script_string() + append
 78                pycharmm.lingo.charmm_script(script)
 79                self._remove_selection(sel)
 80        else:
 81            script = self.create_script_string() + append
 82            pycharmm.lingo.charmm_script(script)
 83
 84        return self

115class WriteScript(CommandScript):
116    """A child of the `CommandScript` class for running CHARMM WRITE command.
117
118    In the constructure a multiline `title` can be provided. The title can later be appended when the `runn()` function is called using its `append` argument.
119    """
120    def __init__(self, filename, title='', **kwargs):
121        super().__init__('write', name=filename, **kwargs)
122        self.filename = filename
123        self.title = ''
124        if title:
125            self.title = '* ' + title + '\n'
126            self.title += '*\n'
127
128    def run(self, append=''):
129        to_append = ''
130        append = append.strip()
131        if append:
132            to_append = append + '\n'
133
134        if self.title:
135            to_append += self.title
136
137        super().run(append=to_append)
138        return self

141def script_factory(command='', required_args=None):
142    """Create a CommandScript object for any CHARMM command.
143
144    Parameters
145    ----------
146    command : str 
147        the CHARMM command to be run (only the name)
148    required_args : list[str]
149        a list of argument names the command requires
150
151    Returns
152    -------
153    UserScript : CommandScript
154        A class with a run method to run the CHARMM command specified
155
156    Examples
157    --------
158    >>> import pycharmm
159
160    To run the GETEnegy command, one can do the following
161    >>> NewGeteClass = pycharmm.script_factory('gete')
162    >>> my_gete = NewGeteClass(...)
163    >>> my_gete.run()
164    
165    """
166    if required_args is None:
167        required_args = []
168
169    required_args = required_args[:]
170
171    class UserScript(CommandScript):
172        def __init__(self, *args, **kwargs):
173            if len(args) > len(required_args):
174                raise ValueError('Too many arguments passed.')
175
176            # At this point all positional arguments are fine.
177            for arg in required_args[len(args):]:
178                if arg not in kwargs:
179                    raise ValueError('Missing value for argument {}.'.format(arg))
180
181            # At this point, all arguments have been passed either as
182            # positional or keyword.
183            if len(required_args) - len(args) != len(kwargs):
184                raise ValueError('Too many arguments passed.')
185
186            for k, v in zip(required_args, args):
187                kwargs[k] = v
188
189            super().__init__(command, **kwargs)
190
191    return UserScript
