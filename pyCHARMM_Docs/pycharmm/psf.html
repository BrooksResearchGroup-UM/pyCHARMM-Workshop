pycharmm.psf

Get data from CHARMM associated with 'psf' files

Functions

  • get_natom -- get the current total number of atoms from CHARMM
  • get_nres -- get the current total number of residues from CHARMM
  • get_nseg -- get the current total number of segments from CHARMM
  • get_ngrp -- get the current total number of groups from CHARMM
  • get_iac -- export a copy of iac (param type codes)
  • get_amass -- export a copy of amass (atom masses)
  • get_charges -- export a copy of cg (atom charges)
  • get_ibase -- export a copy of ibase (last atom of each residue)
  • get_atype -- export a copy of atype (atom name array)
  • get_res -- export a copy of res (residue name array)
  • get_resid -- export a copy of resid (residue identifier array)
  • get_segid -- export a copy of segid (segment identifier array)
  • get_nictot -- export a copy of nictot (nres for each seg)
  • get_igpbs -- export a copy of igpbs (pointer for 1st atom in each group)
  • get_igptyp -- export a copy of gptyp (code type of each group)

Examples

>>> import pycharmm
>>> import pycharmm.psf as psf

Get number of atoms in the system

>>> n = psf.get_natom()

Delete all TIP3 water molecules

>>> water = pycharmm.SelectAtoms(res_name='TIP3')
>>> psf.delete_atoms(water)

104def get_iac():
105    """Get an array of the param type code for each atom
106

161def get_atype():
162    """Get list of IUPAC atom names
163
164    Returns
165    -------
166    atype : string list
167            IUPAC name for each atom
168    """
169    n = get_natom()
170    atype_buffers = [ctypes.create_string_buffer(8) for _ in range(n)]
171    atype_pointers = (ctypes.c_char_p * n)(*map(ctypes.addressof,
172                                                atype_buffers))
173
174    lib.charmm.psf_get_atype(atype_pointers)
175    atype = [b.value.decode(errors='ignore') for b in atype_buffers[0:n]]
176    return atype

299def delete_atoms(selection=None, psort=False):
300    """Delete a selection of atoms from the psf
301
302    Parameters
303    ----------
304    selection : pycharmm.SelectAtoms
305                selection[i] == True <=> atom i is selected
306    psort : boolean
307            True <=> sort psf after deleted atoms are mapped out
308
309    Returns
310    -------
311    status : bool
312             true if successful
313    """
314    if not selection:
315        selection = pycharmm.SelectAtoms().all_atoms()
316
317    sel_c = selection.as_ctypes()
318    isort = ctypes.c_int(psort)
319
320    status = lib.charmm.psf_delete_atoms(sel_c, ctypes.byref(isort))
321
322    status = bool(status)
323    return status

513def get_iblo_inb():
514    """Get non-bonded exclusion list
515
516    Returns
517    -------
518    inb : int list
519          non-bonded exclusion list
520    """
521    natom = get_natom()
522    nnb = get_nnb()
523    iblo = (ctypes.c_int * natom)()
524    inb = (ctypes.c_int * nnb)()
525    lib.charmm.psf_get_iblo_inb(iblo, inb)
526    return list(iblo), list(inb)

529def set_iblo_inb(new_inblo, new_inb):
530    """Set non-bonded exclusion list
531    
532    Parameters
533    ----------
534    new_inblo : int list(natom)
535                INBLO list
536    new_inb : int list(nnb)
537                INB list
538    
539    Returns
540    -------
541    nnb : int
542          number of atom pairs in non-bonded exclusion list
543    """
544    natom = get_natom()
545    nnb = len(new_inb)
546    iblo = (ctypes.c_int * natom)(*new_inblo)
547    inb = (ctypes.c_int * nnb)(*new_inb)
548    nnb = (ctypes.c_int * 1)(nnb)
549    lib.charmm.psf_set_iblo_inb(nnb, iblo, inb)
550    
551    nnb = get_nnb()
552    
553    pycharmm.nbonds.update_bnbnd()
554    pycharmm.image.update_bimag()
555    
556    return nnb
